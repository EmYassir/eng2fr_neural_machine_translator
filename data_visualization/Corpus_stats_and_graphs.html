def crunch_stats(file, token):
    """
    Tallies up statistics on data corpus
    :param file: path and name of input text file
    :param token: path and name of unilingual trained tokenizer
    :param save_path: path where results are saved
    :return: num_seqs (int) the number of sequences,
             num_words (int list) a list of the number of words per sequence,
             num_tokens (int list) as list of the number of tokens per sequence
    """

    # Get total number of sequences in file
    with open(file, "r", encoding="utf-8") as text_file:
        num_seqs = sum([1 for _ in text_file.readlines()])

    with open(file, encoding="UTF-8") as file:
        text_lines = file.readlines()

    # list of string sequences
    examples = []
    # list of integers : number of words per sequence
    num_words = []
    for text_line in text_lines:
        # Leading whitepsace are removed
        line = text_line.strip()
        # Split line per number of white spaces: estimate number of words
        # Append number of words to num_words
        examples.append(line)
        num_words.append(len(line.split()))

    # load trained tokenizer
    file_tokenizer = tfds.features.text.SubwordTextEncoder.load_from_file(token)

    # list of integers : number of tokens per sequence
    num_tokens = []
    for example in examples:
        num_tokens.append(len(file_tokenizer.encode(example)))

    return num_seqs, examples, num_words, num_tokens
